class Solution{
  public:
  
    //Function to return the name of candidate that received maximum votes.
    vector<string> winner(string arr[],int n)
    {
        map<string,int>mp; for(int i=0;i<n;i++) mp[arr[i]]++;
        int c=0;
        for(auto i:mp) c=max(c,i.second);
        vector<string> ans,a; for(auto i:mp){
            if(i.second==c) ans.push_back(i.first);
        }
        sort(ans.begin(),ans.end()); a.push_back(ans[0]); a.push_back(to_string(c));
        return a;
    }
};

        cin>>arr[i];
        Solution obj;
        vector<string> result = obj.winner(arr,n);
