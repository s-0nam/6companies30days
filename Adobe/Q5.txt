class Solution{
    public:
    int m=1e9+7;   
    int numOfWays(int n, int x)
    {
        long long dp[n+1]; memset(dp,0,sizeof(dp));
        dp[0]=dp[1]=1; 
        int mx=pow(n,1.0/x);
        for(int i=2;i<=mx;i++){
            int cur=pow(i,x);
            for(int j=n;j>=cur;j--){
                dp[j]+=dp[j-cur]; dp[j]%=m;
            }
        }
        return dp[n];
    } 
};

    }
