class Solution{
  public:
    int countSubArrayProductLessThanK(const vector<int>& a, int n, long long k) {
        int ans=0; long long unsigned p=1;
        for(int start=0,end=0;end<n;end++){
            p*=a[end];
            while(p>=k){
                p/=a[start++];
            }
            if(p<k) ans+=end-start+1;
        }
        return ans;
    }
};

int main() {
    int t;

    while (t--) {
        int n, i;
        long long int k;
        cin >> n >> k;
        vector<int> arr(n);
        for (i = 0; i < n; i++) cin >> arr[i];
        Solution obj;
        cout << obj.countSubArrayProductLessThanK(arr, n, k) << endl;
    }
    return 0;
}
